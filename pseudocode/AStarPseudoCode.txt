	closed = {}

	start.g = 1
	start.h = manhattanDistance(start)
	start.f = start.g + start.h

	open.add(start)

			done

		close.add(current)

		for all neighbors of current

			neighbor.h = manhattanDistance(neighbor)
			neighbor.f = neighbor.g + neighbor.h

					open.add(neighbor)
				else if (current.g + 1) >= neighbor.g

				end


			end
		end
	end
